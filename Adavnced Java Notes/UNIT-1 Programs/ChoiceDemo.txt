import java.awt.*;
class ChoiceDemo extends Frame
{
	ChoiceDemo()
	{
		FlowLayout f1=new FlowLayout();
		setLayout(f1);
		Choice c1=new Choice();
		c1.add("C Lang");
		c1.add("C++ Lang");
		c1.add("Java Lang");
		c1.add("PHP Lang");
		c1.add("Python Lang");

	}
	public static void main(String args[])
	{
		ChoiceDemo f1=new ChoiceDemo();
		f1.setVisible(true);
		f1.setSize(500,500);
		f1.setTitle("Choice Control");
	}
}
